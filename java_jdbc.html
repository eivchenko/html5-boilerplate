  public Vector<Link> table = new Vector<Link>();

  public  void run(){

    try{

      System.out.println("\n Started router # 0 waiting for connection on port 0" );

      requestSocket = new Socket("localhost", port); //connect to Router 1 on port 1
      out = new ObjectOutputStream(requestSocket.getOutputStream());
      in = new ObjectInputStream(requestSocket.getInputStream());

      System.out.println(" Connected to router # " + port + " on port " + port + "\n");
      BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

      System.out.println(" +++ Starting the exchange");
      sendMessage(table);
    }
    catch(UnknownHostException unknownHost){
      System.err.println("You are trying to connect to an unknown host!");
    }
    catch(Exception exception){
      exception.printStackTrace();
    }
    finally{
      //4: Closing connection
      try{
        System.out.println( "\n Disconnecting");
        in.close();
        out.close();
        requestSocket.close();
      }
      catch(IOException ioException){
        ioException.printStackTrace();
      }
    }
  }

  public void sendMessage(Object strRequest) {
    try{
      out.writeObject(strRequest);
      out.flush();
    }
    catch(IOException ioException){
      ioException.printStackTrace();
    }
  }
